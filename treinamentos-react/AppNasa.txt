import React, {Component} from 'react';

import {
  View,
  Text,
  Image,
  Dimensions,
  FlatList,
  StatusBar,
  StyleSheet,
  ImageBackground,
  TextInput,
  Picker,
} from 'react-native';

const marginTop = StatusBar.height;
const width = Dimensions.get('screen').width;
let arrayholder = [];

class App extends Component {
  constructor(props) {
    super(props);
    this.state = {
      data: [],
      error: null,
      word:null,
    };
  }

  componentDidMount() {
    this.fetchData();
  }

  fetchData = () => {
    const url =
      'https://images-api.nasa.gov/search?q=' + this.state.word;
    this.setState({loading: true});
    fetch(url)
      .then(res => res.json())
      .then(result => {
        if (result.collection.items.length > 0) {
          this.setState({
            data: result.collection.items,
            error: result.error || null,
          });
          this.arrayholder = result.collection.items;
        }
      })
      .catch(error => {
        this.state({error});
      });
  };

  handleSearch = text => {
    if (text.length > 3) {
      const newData = this.arrayholder.filter(item => {
        const actualItemText = item.data[0].title.toUpperCase();
        const textData = text.toUpperCase();
        return actualItemText.indexOf(textData) > -1;
      });
      if (newData.length > 0) {
        this.setState({data: newData});
      }
    }
  };

  renderHeader = () => {
    return (
      <View style={{flexDirection: 'row', marginBottom: 20}}>
        <TextInput
          placeholder="Search"
          lightTheme
          round
          onChangeText={text => this.handleSearch(text)}
          autoCorrect={false}
          style={styles.searchField}
        />        
      </View>
    );
  };

  render() {
    return (
      <View style={styles.mainContainer}>
        <FlatList
          data={this.state.data}
          renderItem={({item}) => (
            <View style={styles.photoContainer}>
              <View style={styles.headerContainer}>
                <Image
                  source={require('./assets/images/nasa-logo.png')}
                  style={styles.userImage}
                />
                <Text style={{marginLeft: 10}}>{item.data[0].title}</Text>
              </View>
              <ImageBackground
                style={styles.image}
                source={require('./assets/images/placeholder.png')}>
                <Image
                  source={{uri: item.links[0].href}}
                  style={styles.image}
                />
              </ImageBackground>
              <Text style={styles.description}>{item.data[0].description}</Text>
              <View>
                <Text style={styles.metaInfo}>
                  Date: {item.data[0].date_created}. Center:{' '}
                  {item.data[0].center}
                </Text>
              </View>
            </View>
          )}
          keyExtractor={(item, index) => index.toString()}
          ListHeaderComponent={this.renderHeader}
        />
      </View>
    );
  }
}

const styles = StyleSheet.create({
  mainContainer: {
    marginTop: 20,
    marginLeft: 5,
    marginRight: 5,
  },

  photoContainer: {
    backgroundColor: '#f0f0f0',
    padding: 10,
    marginBottom: 10,
    borderRadius: 4,
  },

  headerContainer: {
    flexDirection: 'row',
    alignItems: 'center',
    marginBottom: 5,
  },

  image: {
    width: width - 40,
    height: width - 40,
    resizeMode: 'cover',
    borderRadius: 4,
  },

  userImage: {
    width: 60,
    height: 60,
    resizeMode: 'cover',
    borderRadius: 5,
  },

  description: {
    fontSize: 11,
    color: '#333',
    lineHeight: 18,
    marginTop: 10,
    marginBottom: 10,
  },

  metaInfo: {
    fontSize: 10,
    color: '#666',
    textAlign: 'right',
  },

  searchField: {
    borderBottomColor: '#22418d',
    borderBottomWidth: 2,
    flex: 1,
    width: width / 2 - 30,
  },
});

export default App;
